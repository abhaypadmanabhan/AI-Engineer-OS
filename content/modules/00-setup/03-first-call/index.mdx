Make the same call to three frontier models. Measure cost and latency. Compare.

```python file=bench.py colab=content/modules/00-setup/03-first-call/lab.ipynb
import os, time, statistics as stats
from dotenv import load_dotenv
from anthropic import Anthropic
from openai import OpenAI

load_dotenv()
PROMPT = "In one sentence, why do LLMs hallucinate?"

def bench(name, call, n=20):
    lats, outs, costs = [], [], []
    for _ in range(n):
        t = time.perf_counter()
        text, cost = call()
        lats.append((time.perf_counter() - t) * 1000)
        outs.append(text); costs.append(cost)
    return {
        "model": name,
        "p50_ms": round(stats.median(lats)),
        "p95_ms": round(sorted(lats)[int(0.95 * n)]),
        "avg_cost": round(sum(costs) / n, 5),
        "sample": outs[0][:120],
    }

def anthropic_call():
    c = Anthropic()
    r = c.messages.create(
        model="claude-sonnet-4-6",
        max_tokens=80, temperature=0,
        messages=[{"role": "user", "content": PROMPT}],
    )
    # Sonnet 4.6: $3/1M in, $15/1M out
    cost = r.usage.input_tokens * 3/1e6 + r.usage.output_tokens * 15/1e6
    return r.content[0].text, cost

def openai_call():
    c = OpenAI()
    r = c.chat.completions.create(
        model="gpt-5",
        max_tokens=80, temperature=0,
        messages=[{"role": "user", "content": PROMPT}],
    )
    # GPT-5: $5/1M in, $15/1M out (placeholder — check current pricing)
    cost = r.usage.prompt_tokens * 5/1e6 + r.usage.completion_tokens * 15/1e6
    return r.choices[0].message.content, cost

def openrouter_gemini_call():
    c = OpenAI(base_url="https://openrouter.ai/api/v1",
               api_key=os.environ["OPENROUTER_API_KEY"])
    r = c.chat.completions.create(
        model="google/gemini-2.5-pro",
        max_tokens=80, temperature=0,
        messages=[{"role": "user", "content": PROMPT}],
    )
    cost = r.usage.prompt_tokens * 1.25/1e6 + r.usage.completion_tokens * 5/1e6
    return r.choices[0].message.content, cost

if __name__ == "__main__":
    for row in [
        bench("claude-sonnet-4-6", anthropic_call),
        bench("gpt-5",            openai_call),
        bench("gemini-2.5-pro",   openrouter_gemini_call),
    ]:
        print(row)
```

## What this proves

You can swap providers in one line. You can measure their actual cost. You can measure their actual latency. Every downstream decision in this course — "should I use a cheaper model here?" — is grounded in numbers you pulled from your own terminal, not marketing pages.
